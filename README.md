### Gradle
```
allprojects {
    repositories {
        ...
        maven { url 'https://jitpack.io' }
    }
}

dependencies {
        implementation 'com.github.inrhor:pier:0.1.0'
}
```
### Maven
```
<repositories>
    <repository>
        <id>jitpack.io</id>
        <url>https://jitpack.io</url>
    </repository>
</repositories>

<dependency>
    <groupId>com.github.inrhor</groupId>
    <artifactId>pier</artifactId>
    <version>0.1.0</version>
</dependency>
```
